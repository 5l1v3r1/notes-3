TMUX
terminal multiplexer

START
tmux new -s sessname
tmux a #
	-attach
tmux a -t sessname
	-attach to
tmux ls
	-list sessions
tmux kill-session -t sessname
setw -g mode-keys vi
setw -g mode-mouse off
setw -g monitor-activity on
set -g visual-activity on
tmux source-file ~/.tmux.conf
	--reload tmux configuration

Run tmux by default when launching bash shell tty instance:
[[ $TERM != "screen" ]] && [[ ! "$TERM" =~ tmux ]] && [ -z "$TMUX" ] && exec tmux
	--add to .bashrc prior to aliases line
	--checks that bash is not being executed within an existing terminal multiplexer(tmux|screen) context, does not check for tty


~/.tmux.conf

#set default shell for tmux to load with to bash
set-option -g default-shell /bin/bash
set-option -g default-command "/bin/bash -i"
#increase history scrollback per-pane
set-option -g history-limit 50000
#per window
tmux set-option history-limit 5000 \; new-window
#per session
tmux set-option -g history-limit 5000 \; new-session


# reload config file (change file location to your the tmux.conf you want to use)
bind r source-file ~/.tmux.conf

# remap prefix from 'C-b' to 'C-a'
unbind C-b
set-option -g prefix C-a
bind-key C-a send-prefix

# split panes using | and -
bind | split-window -h
bind - split-window -v
unbind '"'
unbind %

# Enable mouse mode (tmux 2.1 and above)
set-option -g mouse on
#<2.1
# Enable mouse control (clickable windows, panes, resizable panes)
set-option -g mouse-select-window on
set-option -g mouse-select-pane on
set-option -g mouse-resize-pane on

# switch panes using Alt-arrow without prefix
bind -n M-Left select-pane -L
bind -n M-Right select-pane -R
bind -n M-Up select-pane -U
bind -n M-Down select-pane -D

# don't rename windows automatically
set-option -g allow-rename off
