IWCONFIG
	jack your antenna power/range
ifconfig wlan0
iw reg set BO
	-set iw regulations to bolivia
ifconfig wlan0 up
iwconfig wlan0 channel 13
iwconfig wlan0 txpower 30
	-now jack up the power of your wireless card
